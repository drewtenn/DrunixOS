# Drunix Pi 5 boot config.
#
# Drop this file on the FAT32 boot partition of a Raspberry Pi 5 SD card,
# alongside Drunix's `kernel8.img` (produced by
# `make ARCH=arm64 PLATFORM=raspi5 kernel8.img`) and the vendor firmware
# blobs from raspberrypi/firmware on GitHub (see boot-pi5/README.md for
# which files and where to fetch them).
#
# Boot a USB-to-UART adapter on the 40-pin GPIO header pins 8 (TXD),
# 10 (RXD), and any ground pin to see the kernel banner over RP1 UART0.

arm_64bit=1
kernel=kernel8.img
enable_uart=1
uart_2ndstage=1

# M7: HDMI framebuffer bring-up via the legacy VC4 property mailbox.
# Caps firmware framebuffer creation at one even if a second HDMI
# head is populated, requests a fixed 1024x768x32 mode so a slow
# monitor handshake at boot does not produce a different geometry,
# and ignores the alpha channel so XRGB8888 reads back as a clean
# opaque image. Drunix's arm64_video_init asks the firmware for the
# same geometry via SET_PHYSICAL_SIZE / SET_DEPTH; these knobs are
# a belt-and-braces layer that matches the firmware's pre-allocated
# framebuffer when config.txt is read first, which makes HDMI come
# up the same way on every cold boot regardless of monitor
# handshake timing.
framebuffer_width=1024
framebuffer_height=768
framebuffer_depth=32
framebuffer_ignore_alpha=1
hdmi_force_hotplug=1
max_framebuffers=1
