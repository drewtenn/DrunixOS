# ARM64 VGA-Style Display Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an ARM64 display mode that gives the `ARCH=arm64` target a visible VGA-style text console comparable to the x86 fallback console.

**Architecture:** ARM64 cannot use PC VGA memory at `0xb8000`, so this feature should implement a VGA-style cell console on top of a Raspberry Pi/QEMU framebuffer. Reuse the existing `kernel/gui` font, color, framebuffer, and display primitives after splitting x86 Multiboot discovery out of the generic framebuffer renderer.

**Tech Stack:** Freestanding C, AArch64 QEMU `raspi3b`, VideoCore mailbox framebuffer property interface, existing `kernel/gui` framebuffer/text renderer, GNU Make, Python boot checks.

## Design Notes

- Do not try to emulate literal VGA hardware on ARM64.
- Do not reuse `kernel/kernel.c`; it is still x86 boot code and contains PC port I/O cursor operations.
- Keep serial as the reliable early console and mirror later output to the framebuffer console once video is initialized.
- Keep ARM64 keyboard/input out of scope for this slice. The first ARM64 VGA-style mode should display the boot log, terminal banner, prompt, and process output while input still arrives over serial.
- Reuse the x86 visual model: 8x16 glyphs, VGA attributes, 16-color palette, scrolling text grid.

```python
#!/usr/bin/env python3
from pathlib import Path
import os
import socket
import subprocess
import time

ROOT = Path(__file__).resolve().parents[1]
LOG_DIR = ROOT / "logs"
SERIAL_LOG = LOG_DIR / "serial-arm64-vga.log"
QEMU_ERR = LOG_DIR / "qemu-arm64-vga.stderr"
MONITOR = LOG_DIR / "qemu-arm64-vga.monitor"
PPM = LOG_DIR / "arm64-vga.ppm"
QEMU = os.environ.get("QEMU_ARM", "qemu-system-aarch64")
MACHINE = os.environ.get("QEMU_ARM_MACHINE", "raspi3b")

REQUIRED_SERIAL = (
    "ARM64 framebuffer console enabled",
    "Drunix ARM64 console",
    "drunix> ",
)


def read_text(path):
    try:
        return path.read_text(errors="ignore")
    except FileNotFoundError:
        return ""


def ppm_has_nonblack_pixels(path):
    data = path.read_bytes()
    if not data.startswith(b"P6\n"):
        return False
    header_end = data.find(b"\n255\n")
    if header_end < 0:
        return False
    pixels = data[header_end + len(b"\n255\n") :]
    return any(byte != 0 for byte in pixels)


def monitor_command(command):
    deadline = time.time() + 5
    while time.time() < deadline:
        try:
            with socket.socket(socket.AF_UNIX, socket.SOCK_STREAM) as sock:
                sock.connect(str(MONITOR))
                sock.sendall(command.encode("ascii") + b"\n")
                return
        except OSError:
            time.sleep(0.1)
    raise SystemExit("timed out connecting to QEMU monitor")


def main():
    build = subprocess.run(
        ["make", "ARCH=arm64", "ARM64_VGA=1", "build"],
        cwd=ROOT,
        capture_output=True,
        text=True,
    )
    if build.returncode != 0:
        raise SystemExit(build.stderr or build.stdout)

    LOG_DIR.mkdir(exist_ok=True)
    for path in (SERIAL_LOG, QEMU_ERR, MONITOR, PPM):
        path.unlink(missing_ok=True)

    with QEMU_ERR.open("w") as err:
        proc = subprocess.Popen(
            [
                QEMU,
                "-M", MACHINE,
                "-kernel", "kernel-arm64.elf",
                "-serial", "null",
                "-serial", f"file:{SERIAL_LOG}",
                "-monitor", f"unix:{MONITOR},server,nowait",
                "-no-reboot",
            ],
            cwd=ROOT,
            stdout=subprocess.DEVNULL,
            stderr=err,
        )

    try:
        deadline = time.time() + 20
        while time.time() < deadline:
            text = read_text(SERIAL_LOG)
            if all(marker in text for marker in REQUIRED_SERIAL):
                break
            if proc.poll() is not None:
                raise SystemExit("QEMU exited before ARM64 VGA markers appeared")
            time.sleep(0.25)
        else:
            missing = [m for m in REQUIRED_SERIAL if m not in read_text(SERIAL_LOG)]
            raise SystemExit("missing ARM64 VGA markers: " + ", ".join(missing))

        monitor_command(f"screendump {PPM}")
        deadline = time.time() + 5
        while time.time() < deadline and not PPM.exists():
            time.sleep(0.1)
        if not PPM.exists() or not ppm_has_nonblack_pixels(PPM):
            raise SystemExit("ARM64 VGA screendump is missing or blank")
        print("arm64 vga console check passed")
    finally:
        if proc.poll() is None:
            monitor_command("quit")
        try:
            proc.wait(timeout=2)
        except subprocess.TimeoutExpired:
            proc.kill()
            proc.wait()


if __name__ == "__main__":
    raise SystemExit(main())
```

- [ ] **Step 2: Implement VGA-style text behavior**

Implement `kernel/console/fb_text_console.c` with:

```c
#include "fb_text_console.h"
#include "kstring.h"

#define FB_TEXT_DEFAULT_ATTR 0x0f

static void fb_text_console_present_cell(fb_text_console_t *console,
                                         uint32_t col,
                                         uint32_t row)
{
	gui_display_present_rect_to_framebuffer(&console->display,
	                                        console->fb,
	                                        (int)col,
	                                        (int)row,
	                                        1,
	                                        1);
}

static void fb_text_console_scroll(fb_text_console_t *console)
{
	uint32_t cols = console->cols;
	uint32_t rows = console->rows;

	k_memmove(console->cells,
	          console->cells + cols,
	          (rows - 1u) * cols * sizeof(console->cells[0]));
	for (uint32_t col = 0; col < cols; col++) {
		gui_cell_t *cell = &console->cells[(rows - 1u) * cols + col];
		cell->ch = ' ';
		cell->attr = console->attr;
	}
	console->cursor_row = rows - 1u;
	console->cursor_col = 0;
	gui_display_present_to_framebuffer(&console->display, console->fb);
}

static void fb_text_console_newline(fb_text_console_t *console)
{
	console->cursor_col = 0;
	if (console->cursor_row + 1u >= console->rows)
		fb_text_console_scroll(console);
	else
		console->cursor_row++;
}

static void fb_text_console_putc(fb_text_console_t *console, char ch)
{
	gui_cell_t *cell;

	if (ch == '\r') {
		console->cursor_col = 0;
		return;
	}
	if (ch == '\n') {
		fb_text_console_newline(console);
		return;
	}
	if (ch == '\b') {
		if (console->cursor_col > 0)
			console->cursor_col--;
		cell = &console->cells[console->cursor_row * console->cols +
		                       console->cursor_col];
		cell->ch = ' ';
		cell->attr = console->attr;
		fb_text_console_present_cell(console,
		                             console->cursor_col,
		                             console->cursor_row);
		return;
	}
	if ((unsigned char)ch < 0x20 || (unsigned char)ch > 0x7e)
		ch = ' ';

	cell = &console->cells[console->cursor_row * console->cols +
	                       console->cursor_col];
	cell->ch = ch;
	cell->attr = console->attr;
	fb_text_console_present_cell(console,
	                             console->cursor_col,
	                             console->cursor_row);
	console->cursor_col++;
	if (console->cursor_col >= console->cols)
		fb_text_console_newline(console);
}
```

Complete the file with `fb_text_console_init()`, `fb_text_console_ready()`, `fb_text_console_write()`, and `fb_text_console_clear()`:

```c
int fb_text_console_init(fb_text_console_t *console,
                         framebuffer_info_t *fb,
                         gui_cell_t *cells,
                         uint32_t cell_capacity)
{
	uint32_t needed;

	if (!console || !fb || !cells)
		return -1;
	if (fb->cell_cols == 0 || fb->cell_rows == 0)
		return -1;
	needed = fb->cell_cols * fb->cell_rows;
	if (needed == 0 || cell_capacity < needed)
		return -1;

	console->cells = cells;
	console->fb = fb;
	console->cols = fb->cell_cols;
	console->rows = fb->cell_rows;
	console->cursor_col = 0;
	console->cursor_row = 0;
	console->attr = FB_TEXT_DEFAULT_ATTR;
	console->ready = 1;
	gui_display_init(&console->display,
	                 cells,
	                 (int)console->cols,
	                 (int)console->rows,
	                 console->attr);
	fb_text_console_clear(console);
	return 0;
}

int fb_text_console_ready(const fb_text_console_t *console)
{
	return console && console->ready;
}

void fb_text_console_write(fb_text_console_t *console,
                           const char *buf,
                           uint32_t len)
{
	if (!fb_text_console_ready(console) || !buf)
		return;
	for (uint32_t i = 0; i < len; i++)
		fb_text_console_putc(console, buf[i]);
}

void fb_text_console_clear(fb_text_console_t *console)
{
	if (!fb_text_console_ready(console))
		return;
	gui_display_fill_rect(&console->display,
	                      0,
	                      0,
	                      (int)console->cols,
	                      (int)console->rows,
	                      ' ',
	                      console->attr);
	console->cursor_col = 0;
	console->cursor_row = 0;
	gui_display_present_to_framebuffer(&console->display, console->fb);
}
```
